API Reference

Complete reference for HesabPay API endpoints with code examples

Endpoints

Create Payment Session

POST

Create a new payment session for single or multiple items

POST https://api.hesab.com/api/v1/payment/create-session

Request

Headers

Authorization: API-KEY your_api_key
Content-Type: application/json
Accept: application/json

Request Body

{
  "email": "customer@example.com",        // optional
  "items": [
    {
      "id": "item1",
      "name": "Product Name",
      "price": 45
    }
  ],
  "redirect_success_url": "https://yoursite.com/success",  // optional
  "redirect_failure_url": "https://yoursite.com/failure"   // optional
}

Response

Success Response (200)

{
  "status_code": 10,
  "success": true,
  "message": "Payment session created successfully",
  "session_id": "sess_1234567890",
  "payment_url": "https://checkout.hesabpay.com/pay/sess_1234567890",
  "expires_at": "2024-02-12T12:04:28Z"
}

Code Examples

import requests
import json

def create_payment_session(api_key, items, email=None, redirect_urls=None):
    """
    Create a payment session with HesabPay
    
    Args:
        api_key (str): Your API key
        items (list): List of items with id, name, and price
        email (str, optional): Customer email
        redirect_urls (dict, optional): Success and failure URLs
    
    Returns:
        dict: API response with session details
    """
    endpoint = "https://api.hesab.com/api/v1/payment/create-session"
    
    headers = {
        "Authorization": f"API-KEY {api_key}",
        "Content-Type": "application/json",
        "Accept": "application/json"
    }
    
    payload = {
        "items": items
    }
    
    # Add optional parameters
    if email:
        payload["email"] = email
    
    if redirect_urls:
        payload.update(redirect_urls)
    
    try:
        response = requests.post(endpoint, json=payload, headers=headers)
        response.raise_for_status()
        return response.json()
    
    except requests.exceptions.RequestException as e:
        return {"error": "Request failed", "message": str(e)}

# Example usage
api_key = "your_api_key_here"
items = [
    {"id": "item1", "name": "Product 1", "price": 45},
    {"id": "item2", "name": "Product 2", "price": 20}
]

result = create_payment_session(
    api_key=api_key,
    items=items,
    email="customer@example.com",
    redirect_urls={
        "redirect_success_url": "https://yoursite.com/success",
        "redirect_failure_url": "https://yoursite.com/failure"
    }
)

print(json.dumps(result, indent=2))
